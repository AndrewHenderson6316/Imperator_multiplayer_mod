dip_offer_truce = {
	scope = country

	saved_scopes = {
		target
	}

	#confirm_title = "make_player_subject"
	#confirm_text = "make_player_confirm" 

	ai_is_valid = {
		always = no
	}
	is_valid = {
		in_diplomatic_range = scope:target
		NOT = {
			scope:target = root
			is_ai = yes
		}
		is_subject = no 
	}

	effect = {
		clear_variable_list = bilateral_truce
		set_variable = {
			name = bilateral_truce
			value = 0
		}
		save_scope_as = sender
		set_variable = {
		  name = send_to
		  value = scope:target
		}
	}
}
