

make_player_subject = {
	scope = country

	saved_scopes = {
		target
	}
	ai_is_valid = {
		always = no
	}
	is_valid = {
		in_diplomatic_range = scope:target
		NOT = {
			scope:target = {
				this = root
				is_subject = yes 
			}
		}
	}

	effect = {
		save_scope_as = sender

		set_variable = {
			name = subject_to_be
			value = scope:target
		}

	}
}
