make_player_subject = {
	scope = country

	saved_scopes = {
		target
	}

	#confirm_title = "make_player_subject"
	#confirm_text = "make_player_confirm" 

	ai_is_valid = {
		always = no
	}
	is_valid = {
		in_diplomatic_range = scope:target
		NOT = {
			scope:target = root
			is_ai = yes
		}
		is_subject = no 
	}
	effect = {
		save_scope_as = sender
		set_variable = {
		  name = send_to
		  value = scope:target
		}

	}
}

	}

	ai_is_valid = {

	}
	is_valid = {
		var:send_to = {
			num_of_cities <= 50

		}

        is_tribal = no

	}

		save_scope_as = sender
		var:send_to = {
			trigger_event = {
				id = player_subject_events.2
			}
		}

	}
